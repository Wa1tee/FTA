\subsection{Laskuesimerkki} \cite{seppanen2006maol}
\begin{center}
\begin{align*}
	E_{alussa} 		&= E_{lopussa} \\
	E_{K1} + E_{P1} &= E_{K2} + E_{P2} \\
	\dfrac{1}{2}mv^2_1 + mgh_1 &= \dfrac{1}{2}mv^2_2 + mgh_2 \\
	\dfrac{1}{2}mv^2_1 &= \dfrac{1}{2}mv^2_2 + mgh_2 - mgh_1 \\
	v_1 &= \sqrt{v^2_2 + 2g(h_2 - h_1)} \\
	&= \sqrt{(1,89\dfrac{m}{s^2})^2 + 2 \cdot 9,81\dfrac{m}{s^2}(0,27 m - 0,12 m)} \\
	&= 6,5141 \dfrac{m}{s^2}\approx 6,5\dfrac{m}{s^2} \\
\end{align*}
\end{center}
